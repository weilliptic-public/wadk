Table: invoice
Description: Stores invoices for customer purchases.
Columns:
  - invoice_id (INTEGER, NOT NULL, PK)
  - customer_id (INTEGER, NOT NULL, FK → customer.customer_id)
  - invoice_date (DATETIME, NOT NULL)
  - billing_address (NVARCHAR(70), NULL)
  - billing_city (NVARCHAR(40), NULL)
  - billing_state (NVARCHAR(40), NULL)
  - billing_country (NVARCHAR(40), NULL)

DATABASE: chinook
Table: playlist
Description: Stores playlists of tracks.

  - playlist_id (INTEGER, NOT NULL, PK)

DATABASE: chinook
Table: playlist_track
Description: Join table for many-to-many relationship between playlists and tracks.
Columns:
  - playlist_id (INTEGER, NOT NULL, FK → playlist.playlist_id)
  - track_id (INTEGER, NOT NULL, FK → track.track_id)

DATABASE: chinook
Table: track
Description: Stores music tracks.
Columns:
  - track_id (INTEGER, NOT NULL, PK)
  - name (NVARCHAR(200), NOT NULL)
  - album_id (INTEGER, NULL, FK → album.album_id)
  - media_type_id (INTEGER, NOT NULL, FK → media_type.media_type_id)
  - genre_id (INTEGER, NULL, FK → genre.genre_id)
  - composer (NVARCHAR(220), NULL)
  - milliseconds (INTEGER, NOT NULL)
  - bytes (INTEGER, NULL)
  - unit_price (NUMERIC(10,2), NOT NULL)
